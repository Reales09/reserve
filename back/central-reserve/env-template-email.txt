# Configuración SMTP para envío de emails
# Ejemplo para Gmail con STARTTLS (RECOMENDADO)
SMTP_HOST=smtp.gmail.com
SMTP_PORT=587
SMTP_USER=tu-email@gmail.com
SMTP_PASS=tu-contraseña-de-aplicación
FROM_EMAIL=reservas@trattorialabella.com
SMTP_USE_STARTTLS=true
SMTP_USE_TLS=false

# Configuración de Seguridad:
# - SMTP_USE_STARTTLS=true: Usa STARTTLS (puerto 587) - RECOMENDADO
# - SMTP_USE_TLS=true: Usa TLS directo (puerto 465)
# - Si ambos son false: Envío sin cifrado (NO RECOMENDADO para producción)

# Nota: Para Gmail, necesitas usar una "Contraseña de aplicación" en lugar de tu contraseña normal
# Para generar una: https://myaccount.google.com/apppasswords
